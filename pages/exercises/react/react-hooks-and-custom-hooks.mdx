## Coding Challenges

### 1. Simple useState Hook

Create a `SwapColors` component using the `useState` hook to toggle between two colors when a button "Swap Colors" is clicked.

**Tips:** Colors should be toggled between "red" and "blue". No need to import React. You must use the `useState` hook like this: `React.useState()`.

<CodingChallenge
  initialCode={`
    function SwapColors() {
      let color; // This is to be replaced by the useState code.

      return (
        <div>
          {/* Width and Height should be 100 */}
          <div style={{ width: 0, height: 0, backgroundColor: color }}></div>
        </div>
      );
    }

    // Render the component
    render(<SwapColors />);

`}
  solution={`
function SwapColors() {
const [color, setColor] = React.useState("red");

      const toggleColors = () => {
        setColor(color === "red" ? "blue" : "red");
      };

      return (
        <div>
          <div style={{ width: 100, height: 100, backgroundColor: color }}></div>
          <button onClick={toggleColors}>Swap Colors</button>
        </div>
      );
    }

    // Render the component
    render(<SwapColors />);

`}
/>

### 2. Basic useEffect Hook

Create a `Timer` component that uses `useEffect` to update and display the current time every second.

<CodingChallenge
  initialCode={`
function Timer() {
  // Your code here
}

// Render the component
render(<Timer />);
`}
  solution={`
function Timer() {
const [time, setTime] = React.useState(new Date().toLocaleTimeString());

React.useEffect(() => {
const timer = setInterval(() => {
setTime(new Date().toLocaleTimeString());
}, 1000);

    return () => clearInterval(timer);

}, []);

return <div>Current Time: {time}</div>;
}

// Render the component
render(<Timer />);
`}
/>

### 3. Simple Custom Hook

Create a custom hook called `useToggle` that manages a boolean state and provides a function to toggle it. Then, use this hook in a component.

<CodingChallenge
  initialCode={`
// Create your custom hook here

function ToggleComponent() {
// Use your custom hook here
}

// Render the component
render(<ToggleComponent />);
`}
  solution={`
function useToggle(initialValue) {
const [value, setValue] = React.useState(initialValue);

      const toggle = () => setValue(!value);

      return [value, toggle];
    }

function ToggleComponent() {
const [isOn, toggleIsOn] = useToggle(false);

return (

<div>
  <p>{isOn ? "ON" : "OFF"}</p>
  <button onClick={toggleIsOn}>Toggle</button>
</div>
); }

// Render the component
render(<ToggleComponent />);
`}
/>
